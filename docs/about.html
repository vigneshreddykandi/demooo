const form = document.querySelector('form'); const urlInput = document.querySelector('#url'); const resultDiv = document.querySelector('#result'); form.addEventListener('submit', async (e) => { e.preventDefault(); const url = urlInput.value; try { const response = await fetch(`https://api.shrtco.de/v2/shorten?url=${url}`); const data = await response.json(); const shortUrl = data.result.short_link; resultDiv.innerHTML = `

Shortened URL:

${shortUrl} `; } catch (error) { console.error(error); } });
